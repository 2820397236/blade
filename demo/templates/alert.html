demo2


   if (!this.alert2) {
    this.alert2 = new UIAlert({
      datamodel: {
        title: 'demo01',
        content: 'content01'
      },
      okAction: function () {
        alert('ok');
        this.hide();
      },
      cancelAction: function () {
        alert('cancel');
        this.hide();
      }
    });
  }
  this.alert2.show();

demo3


    if (!this.alert3) {
    this.alert3 = new UIAlert({
      datamodel: {
        title: 'demo02',
        content: 'content02',
        btns: [
          { name: '知道了', className: 'cui-btns-ok' },
          { name: '不知道', className: 'cui-btns-no' },
          { name: '未知', className: 'cui-btns-unknown' }
        ]
      },
      events: {
        'click .cui-btns-ok': 'okAction',
        'click .cui-btns-no': 'noAction',
        'click .cui-btns-unknown': 'unknownAction'
      },
      okAction: function () {
        alert('知道了');
        this.hide();
      },
      noAction: function () {
        alert('不知道');
        this.hide();
      },
      unknownAction: function () {
        alert('未知');
        this.hide();
      }
    });
  }
  this.alert3.show();
