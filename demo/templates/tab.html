tab组件

基本参数说明


this.num = new UISelect({
  //数据模型，会影响插件渲染
  datamodel: {
    data: [],
    curClass: 'cui-tab-current',
    selectedKey: null,
    index: 0
  },
  //选项改变执行的事件
  onChange: function (data) {
    console.log(data);
  };
  //UI组件包裹器，会放到什么容器里面
  wrapper: this.$('.simple_switch')
});

tab简单应用


this.tab = new UITab({
  datamodel: {
    data: [
      { id: 1, name: '中国' },
      { id: 2, name: '美国' },
      { id: 3, name: '日本' }
    ]
  },
  wrapper: this.$el.find('.wrapper')
});
this.tab.show();

setIndex/getIndex


this.tab1 = new UITab({
  datamodel: {
    data: [
      { id: 1, name: '图片' },
      { id: 2, name: '音乐' },
      { id: 3, name: '视频' },
      { id: 4, name: '其它' }
    ],
    curClass: 'cui-tab-current',
    index: 0
  },
  onChange: function() {
    console.log('onChange myself:' + this.el.attr('data-key')+ this.el.html() );
  },
  wrapper: this.$el.find('.wrapper1')
});
this.tab1.show();

//对应按钮事件
getIndex: function() {
  var index = this.tab1.getIndex();
  this.$el.find('.getInfo').val(index);
},
setIndex: function() {
  var index = this.$el.find('.setInfo').val() || 0;
  this.tab1.setIndex(index);
},
