demo1


    if(!this.demo1) {
        var data = [{ name: '价格/星级' },
          { name: '位置区域' },
          { name: '品牌' },
          { name: '测试'}];


        var el = $(e.currentTarget);
        var index = parseInt(Math.random() * 4);
        var dir = (e.pageY > 200) ? 'down' : 'up';
        this.demo1 = new UIBubbleLayer({
          triggerEl: el,
          datamodel: {
            data: data,
            dir: dir,
            index: index
          },
          onClick:  function (e, data, index, el) {
            console.log(arguments);
            this.setIndex(index);
            this.hide();
          }
        });
      }
      this.demo1.show();
